# Test case 1: Integerss (assume all elements type is int):
     >>> my_list = MyList()
     >>> my_list.append(3)
     >>> my_list.append(50)
     >>> my_list.append(-1)
     >>> print(my_list)
     [3, 50, -1]
     >>> my_list.print_sorted()
     [-1, 3, 50]
     >>> print(my_list)
     [3, 50, -1]

     >>> another_list = MyList()
     >>> another_list.append(300)
     >>> another_list.append(50)
     >>> another_list.append(-1)
     >>> print(another_list)
     [300, 50, -1]
     >>> another_list.print_sorted()
     [-1, 50, 300]
     >>> print(another_list)
     [300, 50, -1]
